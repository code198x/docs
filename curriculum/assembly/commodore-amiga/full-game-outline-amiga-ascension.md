### Code Pattern
```asm
; Equipment stat calculation
calculate_player_stats:
    ; Base stats
    lea     base_stats,a0
    lea     current_stats,a1
    move.w  #STAT_COUNT-1,d7
.copy_base:
    move.w  (a0)+,(a1)+
    dbf     d7,.copy_base

    ; Apply weapon stats
    move.w  equipped_weapon,d0
    bmi.s   .no_weapon
    mulu    #WEAPON_SIZE,d0
    lea     weapon_data(pc,d0.l),a0
    lea     current_stats,a1
    add.w   WEAPON_ATK(a0),STAT_ATK(a1)
    add.w   WEAPON_MAG(a0),STAT_MAG(a1)
.no_weapon:

    ; Apply armour stats
    move.w  equipped_armour,d0
    bmi.s   .no_armour
    mulu    #ARMOUR_SIZE,d0
    lea     armour_data(pc,d0.l),a0
    lea     current_stats,a1
    add.w   ARMOUR_DEF(a0),STAT_DEF(a1)
.no_armour:

    ; Apply accessory effects
    jsr     apply_accessory_effects
    rts
```
